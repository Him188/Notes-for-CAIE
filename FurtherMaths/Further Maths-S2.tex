    \section{Binomial distribution}

    \begin{displaymath}
        X \sim \left( n, p \right)
    \end{displaymath}

    \subsection{Conditions for binomial distribution}

    \begin{itemize}
        \item n repeated trials
        \item independent trials
        \item two outcomes
        \item P is a constant
    \end{itemize}

    \subsection{Convert $\geq p$ to $\leq 1-p$}
    \begin{displaymath}
        X \sim \left( n, p \right)
    \end{displaymath}
    \begin{displaymath}
        Y \sim \left( n, 1 - p \right)
    \end{displaymath}
    \begin{displaymath}
        P(X \geq a ) = P(Y \leq  n-a)
    \end{displaymath}
